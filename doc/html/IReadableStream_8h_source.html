00070 template<class T>
00071 IReadableStream& fromStream(IReadableStream& stream, T& buf) throw (StreamException)
00072 {
00073    size_t idx(0);
00074    const size_t sz(sizeof(buf));
00075    while (idx < sz) {
00076       size_t result(stream.read(reinterpret_cast<char*>(&buf) + idx, sz - idx));
00077       if (result <= 0) {
00078          throw StreamException("cannot read from IReadableStream");
00079       }
00080       idx += result;
00081    }
00082    return stream;
00083 }
