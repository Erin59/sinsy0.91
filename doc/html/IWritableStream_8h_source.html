00069 template<class T>
00070 IWritableStream& toStream(IWritableStream& stream, const T& value)
00071 {
00072    size_t idx(0);
00073    const size_t size(sizeof(value));
00074    while (idx < size) {
00075       size_t result(stream.write(reinterpret_cast<const char*>(&value) + idx, size - idx));
00076       if (result <= 0) {
00077          throw StreamException("cannot write to IWritableStream");
00078       }
00079       idx += result;
00080    }
00081    return stream;
00082 }
